# Common environment for all jobs

PROJECT_DIR=/home/truonggiang/ride-hailing
PYTHON=/home/truonggiang/ride-hailing/.venv/bin/python
SHELL=/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin

# -----------------------------------------------------------------------
# 1) Monthly TLC download - 02:00 on the 1st of every month
#    Then validate the latest file. If validation fails, email goes to
#    MAILTO (set above) so you find out before Stage 3 breaks.
# -----------------------------------------------------------------------
0 2 1 * * cd $PROJECT_DIR && $PYTHON scripts/download_tlc.py >> logs/cron.download_tlc.log 2>&1 && $PYTHON scripts/validate_data.py --latest >> logs/cron.validate.log 2>&1

# -----------------------------------------------------------------------
# 2) Hourly weather fetch - top of every hour
# -----------------------------------------------------------------------
0 * * * * cd $PROJECT_DIR && $PYTHON scripts/fetch_weather.py >> logs/cron.fetch_weather.log 2>&1
